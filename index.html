int fib(n) {
if (n <= 1)
return n;
else if (n == 2)
return 1;
else
return fib(n-2) + fib(n-1);
}
void main() {
int num;
scanf("%d",&num);
int array[num], ele;
for( int i=1 ; i <= num ; i++ ) {
ele = fib(i);
array[i-1]=ele;
}
}
